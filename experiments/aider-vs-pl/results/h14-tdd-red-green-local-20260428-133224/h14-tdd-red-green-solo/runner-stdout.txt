function mergeDuplicates(contacts) {
  // Group contacts by email
  const groups = new Map();
  for (const contact of contacts) {
    if (!groups.has(contact.email)) {
      groups.set(contact.email, []);
    }
    groups.get(contact.email).push(contact);
  }

  // Merge each group
  const merged = [];
  for (const [email, group] of groups) {
    const mergedContact = {
      name: null,
      email: email,
      phone: null,
      company: null,
      createdAt: Date.now(),
    };

    // For each field, take the last non-null value from the group
    for (const field of ['name', 'phone', 'company']) {
      for (let i = group.length - 1; i >= 0; i--) {
        if (group[i][field] !== null) {
          mergedContact[field] = group[i][field];
          break;
        }
      }
    }

    merged.push(mergedContact);
  }

  return merged;
}
