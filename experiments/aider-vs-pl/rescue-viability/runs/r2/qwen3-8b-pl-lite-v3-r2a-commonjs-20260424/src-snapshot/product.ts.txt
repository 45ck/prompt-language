interface Product {
  sku: string;
  title: string;
  price: number;
  inStock: boolean;
}

export function createProduct(input: Partial<Product> = {}) {
  return {
    sku: input.sku || '',
    title: input.title || '',
    price: input.price !== undefined ? input.price : 0,
    inStock: input.inStock !== undefined ? input.inStock : true,
  };
}
